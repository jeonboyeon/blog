짝수는 싫어요

❍ 문제 설명

정수 n이 매개변수로 주어질 때, n 이하의 홀수가 오름차순으로 담긴 배열을 return하도록 solution 함수를 완성해주세요.

❍ 제한 사항

1 ≤ n ≤ 100

❍ 입출력 예

n result
10 [1, 3, 5, 7, 9]
15 [1, 3, 5, 7, 9, 11, 13, 15]

❍ 나의 풀이

function solution(n) {
    let answer = [];

    // for문으로 n만큼 i를 0부터 증가시켜줍니다.
    for (let i = 0; i <= n; i++) {
        // if문을 통해 i % 2 가 1과 같다면
        if (i % 2 === 1){
            // 일치하는 i 값들만 answer에 넣어줍니다.
            answer = [...answer, i];
        }
    }
    return answer;
}

❍ 다른 사람의 풀이

function solution(n) {
    var answer = [];

    // for 문을 통해 i를 1부터 n 값과 같아질 때까지 i+=2 값씩 증가시켜주어 그 i 값들을 answer에 push 시켜주셨습니다.
    for (let i = 1; i<=n; i+=2) answer.push(i)

    return answer;
}
