배열의 평균 구하기

❍ 문제 설명

정수를 담고 있는 배열 arr의 평균값을 return하는 함수, solution을 완성해보세요.

❍ 입출력 예

arr return
[1, 2, 3, 4] 2.5
[5, 5] 5

❍ 나의 풀이

function solution(arr) {
    var answer = 0;
    
    // arr 길이를 구하여 각 값들을 answer에 더해준다.
    for(let i=0; i<arr.length; i++){
        answer += arr[i];
    }
    // 각 i의 값을 더하여준 answer을 arr의 총 갯 수 만큼 나누어준다.
    return answer/arr.length;
}

❍ 다른 사람의 풀이

function average(array){
    // arr의 배열을 .reduce() 함수를 사용하여 각각 더해주고, arr 총 갯수만큼 나누어주셨다.
    return array.reduce((a, b) => a + b) / array.length;
}
